1//! Output audio via the OS via mixers or play directly
2//!
3//! This module provides a builder that's used to configure and open audio output. Once
4//! opened sources can be mixed into the output via `DeviceSink::mixer`.
5//!
6//! There is also a convenience function `play` for using that output mixer to
7//! play a single sound.

22/// `cpal::Stream` container. Use `mixer()` method to control output.
23///
24/// <div class="warning">When dropped playback will end, and the associated
25/// OS-Sink will be disposed</div>
26///
27/// # Note
28/// On drop this will print a message to stderr or emit a log msg when tracing is
29/// enabled. Though we recommend you do not you can disable that print/log with:
30/// [`DeviceSink::log_on_drop(false)`](DeviceSink::log_on_drop).
31/// If the `DeviceSink` is dropped because the program is panicking we do not print
32/// or log anything.
33///
34/// # Example
35/// ```no_run
36/// # use rodio::DeviceSinkBuilder;
37/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
38/// let mut handle = DeviceSinkBuilder::open_default_sink()?;
39/// handle.log_on_drop(false); // Not recommended during development
40/// println!("Output config: {:?}", handle.config());
41/// let mixer = handle.mixer();
42/// # Ok(())
43/// # }
44/// ```
45pub struct MixerDeviceSink {
46    config: DeviceSinkConfig,
47    mixer: Mixer,
48    log_on_drop: bool,
49    _stream: cpal::Stream,
50}

70impl Drop for MixerDeviceSink {
71    fn drop(&mut self) {
72        if self.log_on_drop && !std::thread::panicking() {
73            #[cfg(feature = "tracing")]
74            tracing::debug!("Dropping DeviceSink, audio playing through this sink will stop");
75            #[cfg(not(feature = "tracing"))]
76            eprintln!("Dropping DeviceSink, audio playing through this sink will stop, to prevent this message from appearing use tracing or call `.log_on_drop(false)` on this DeviceSink")
77        }
78    }
79}

158/// Convenience builder for audio OS-player.
159/// It provides methods to configure several parameters of the audio output and opening default
160/// device. See examples for use-cases.
161///
162/// <div class="warning">When the DeviceSink is dropped playback will end, and the associated
163/// OS-Sink will be disposed</div>
164pub struct DeviceSinkBuilder<E = fn(cpal::StreamError)>
165where
166    E: FnMut(cpal::StreamError) + Send + 'static,
167{
168    device: Option<cpal::Device>,
169    config: DeviceSinkConfig,
170    error_callback: E,
171}
172
173impl Default for DeviceSinkBuilder {
174    fn default() -> Self {
175        Self {
176            device: None,
177            config: DeviceSinkConfig::default(),
178            error_callback: default_error_callback,
179        }
180    }
181}
182
183impl DeviceSinkBuilder {
184    /// Sets output device and its default parameters.
185    pub fn from_device(device: cpal::Device) -> Result<DeviceSinkBuilder, DeviceSinkError> {
186        let default_config = device
187            .default_output_config()
188            .map_err(DeviceSinkError::DefaultSinkConfigError)?;
189
190        Ok(Self::default()
191            .with_device(device)
192            .with_supported_config(&default_config))
193    }
194
195    /// Sets default OS-Sink parameters for default output audio device.
196    pub fn from_default_device() -> Result<DeviceSinkBuilder, DeviceSinkError> {
197        let default_device = cpal::default_host()
198            .default_output_device()
199            .ok_or(DeviceSinkError::NoDevice)?;
200        Self::from_device(default_device)
201    }
202
203    /// Try to open a new OS-Sink for the default output device with its default configuration.
204    /// Failing that attempt to open OS-Sink with alternative configuration and/or non default
205    /// output devices. Returns stream for first of the tried configurations that succeeds.
206    /// If all attempts fail return the initial error.
207    pub fn open_default_sink() -> Result<MixerDeviceSink, DeviceSinkError> {
208        Self::from_default_device()
209            .and_then(|x| x.open_stream())
210            .or_else(|original_err| {
211                let mut devices = match cpal::default_host().output_devices() {
212                    Ok(devices) => devices,
213                    Err(err) => {
214                        #[cfg(feature = "tracing")]
215                        tracing::error!("error getting list of output devices: {err}");
216                        #[cfg(not(feature = "tracing"))]
217                        eprintln!("error getting list of output devices: {err}");
218                        return Err(original_err);
219                    }
220                };
221                devices
222                    .find_map(|d| {
223                        Self::from_device(d)
224                            .and_then(|x| x.open_sink_or_fallback())
225                            .ok()
226                    })
227                    .ok_or(original_err)
228            })
229    }
230}

232impl<E> DeviceSinkBuilder<E>
233where
234    E: FnMut(cpal::StreamError) + Send + 'static,
235{
236    /// Sets output audio device keeping all existing stream parameters intact.
237    /// This method is useful if you want to set other parameters yourself.
238    /// To also set parameters that are appropriate for the device use [Self::from_device()] instead.
239    pub fn with_device(mut self, device: cpal::Device) -> DeviceSinkBuilder<E> {
240        self.device = Some(device);
241        self
242    }
243
244    /// Sets number of OS-Sink's channels.
245    pub fn with_channels(mut self, channel_count: ChannelCount) -> DeviceSinkBuilder<E> {
246        assert!(channel_count.get() > 0);
247        self.config.channel_count = channel_count;
248        self
249    }
250
251    /// Sets OS-Sink's sample rate.
252    pub fn with_sample_rate(mut self, sample_rate: SampleRate) -> DeviceSinkBuilder<E> {
253        self.config.sample_rate = sample_rate;
254        self
255    }
256
257    /// Sets preferred output buffer size.
258    ///
259    /// To play sound without any glitches the audio card may never receive a
260    /// sample to late. Some samples might take longer to generate then
261    /// others. For example because:
262    ///  - The OS preempts the thread creating the samples. This happens more
263    ///    often if the computer is under high load.
264    ///  - The decoder needs to read more data from disk.
265    ///  - Rodio code takes longer to run for some samples then others
266    ///  - The OS can only send audio samples in groups to the DAC.
267    ///
268    /// The OS solves this by buffering samples. The larger that buffer the
269    /// smaller the impact of variable sample generation time. On the other
270    /// hand Rodio controls audio by changing the value of samples. We can not
271    /// change a sample already in the OS buffer. That means there is a
272    /// minimum delay (latency) of `<buffer size>/<sample_rate*channel_count>`
273    /// seconds before a change made through rodio takes effect.
274    ///
275    /// # Large vs Small buffer
276    /// - A larger buffer size results in high latency. Changes made trough
277    ///   Rodio (volume/skip/effects etc) takes longer before they can be heard.
278    /// - A small buffer might cause:
279    ///   - Higher CPU usage
280    ///   - Playback interruptions such as buffer underruns.
281    ///   - Rodio to log errors like: `alsa::poll() returned POLLERR`
282    ///
283    /// # Recommendation
284    /// If low latency is important to you consider offering the user a method
285    /// to find the minimum buffer size that works well on their system under
286    /// expected conditions. A good example of this approach can be seen in
287    /// [mumble](https://www.mumble.info/documentation/user/audio-settings/)
288    /// (specifically the *Output Delay* & *Jitter buffer*.
289    ///
290    /// These are some typical values that are a good starting point. They may also
291    /// break audio completely, it depends on the system.
292    /// - Low-latency (audio production, live monitoring): 512-1024
293    /// - General use (games, media playback): 1024-2048
294    /// - Stability-focused (background music, non-interactive): 2048-4096
295    pub fn with_buffer_size(mut self, buffer_size: cpal::BufferSize) -> DeviceSinkBuilder<E> {
296        self.config.buffer_size = buffer_size;
297        self
298    }
299
300    /// Select scalar type that will carry a sample.
301    pub fn with_sample_format(mut self, sample_format: SampleFormat) -> DeviceSinkBuilder<E> {
302        self.config.sample_format = sample_format;
303        self
304    }

306    /// Set available parameters from a CPAL supported config. You can get a list of
307    /// such configurations for an output device using [crate::stream::supported_output_configs()]
308    pub fn with_supported_config(
309        mut self,
310        config: &cpal::SupportedStreamConfig,
311    ) -> DeviceSinkBuilder<E> {
312        self.config = DeviceSinkConfig {
313            channel_count: NonZero::new(config.channels())
314                .expect("no valid cpal config has zero channels"),
315            sample_rate: NonZero::new(config.sample_rate())
316                .expect("no valid cpal config has zero sample rate"),
317            sample_format: config.sample_format(),
318            ..Default::default()
319        };
320        self
321    }
322
323    /// Set all OS-Sink parameters at once from CPAL stream config.
324    pub fn with_config(mut self, config: &cpal::StreamConfig) -> DeviceSinkBuilder<E> {
325        self.config = DeviceSinkConfig {
326            channel_count: NonZero::new(config.channels)
327                .expect("no valid cpal config has zero channels"),
328            sample_rate: NonZero::new(config.sample_rate)
329                .expect("no valid cpal config has zero sample rate"),
330            buffer_size: config.buffer_size,
331            ..self.config
332        };
333        self
334    }

336    /// Set a callback that will be called when an error occurs with the stream
337    pub fn with_error_callback<F>(self, callback: F) -> DeviceSinkBuilder<F>
338    where
339        F: FnMut(cpal::StreamError) + Send + 'static,
340    {
341        DeviceSinkBuilder {
342            device: self.device,
343            config: self.config,
344            error_callback: callback,
345        }
346    }
347
348    /// Open OS-Sink using parameters configured so far.
349    pub fn open_stream(self) -> Result<MixerDeviceSink, DeviceSinkError> {
350        let device = self.device.as_ref().expect("No output device specified");
351
352        MixerDeviceSink::open(device, &self.config, self.error_callback)
353    }
354
355    /// Try opening a new OS-Sink with the builder's current stream configuration.
356    /// Failing that attempt to open stream with other available configurations
357    /// supported by the device.
358    /// If all attempts fail returns initial error.
359    pub fn open_sink_or_fallback(&self) -> Result<MixerDeviceSink, DeviceSinkError>
360    where
361        E: Clone,
362    {
363        let device = self.device.as_ref().expect("No output device specified");
364        let error_callback = &self.error_callback;
365
366        MixerDeviceSink::open(device, &self.config, error_callback.clone()).or_else(|err| {
367            for supported_config in supported_output_configs(device)? {
368                if let Ok(handle) = DeviceSinkBuilder::default()
369                    .with_device(device.clone())
370                    .with_supported_config(&supported_config)
371                    .with_error_callback(error_callback.clone())
372                    .open_stream()
373                {
374                    return Ok(handle);
375                }
376            }
377            Err(err)
378        })
379    }
380}

382/// A convenience function. Plays a sound once.
383/// Returns a `Player` that can be used to control the sound.
384pub fn play<R>(mixer: &Mixer, input: R) -> Result<Player, PlayError>
385where
386    R: Read + Seek + Send + Sync + 'static,
387{
388    let input = decoder::Decoder::new(input)?;
389    let player = Player::connect_new(mixer);
390    player.append(input);
391    Ok(player)
392}

423/// Errors that might occur when interfacing with audio output.
424#[derive(Debug, thiserror::Error)]
425pub enum DeviceSinkError {
426    /// Could not start playing the sink, see [cpal::PlayStreamError] for
427    /// details.
428    #[error("Could not start playing the stream")]
429    PlayError(#[source] cpal::PlayStreamError),
430    /// Failed to get the stream config for the given device. See
431    /// [cpal::DefaultStreamConfigError] for details.
432    #[error("Failed to get the config for the given device")]
433    DefaultSinkConfigError(#[source] cpal::DefaultStreamConfigError),
434    /// Error opening sink with OS. See [cpal::BuildStreamError] for details.
435    #[error("Error opening the stream with the OS")]
436    BuildError(#[source] cpal::BuildStreamError),
437    /// Could not list supported configs for the device. Maybe it
438    /// disconnected. For details see: [cpal::SupportedStreamConfigsError].
439    #[error("Could not list supported configs for the device. Maybe its disconnected?")]
440    SupportedConfigsError(#[source] cpal::SupportedStreamConfigsError),
441    /// Could not find any output device
442    #[error("Could not find any output device")]
443    NoDevice,
444    /// New cpal sample format that rodio does not yet support please open
445    /// an issue if you run into this.
446    #[error("New cpal sample format that rodio does not yet support please open an issue if you run into this.")]
447    UnsupportedSampleFormat,
448}

450impl MixerDeviceSink {
451    fn validate_config(config: &DeviceSinkConfig) {
452        if let BufferSize::Fixed(sz) = config.buffer_size {
453            assert!(sz > 0, "fixed buffer size must be greater than zero");
454        }
455    }
456
457    pub(crate) fn open<E>(
458        device: &cpal::Device,
459        config: &DeviceSinkConfig,
460        error_callback: E,
461    ) -> Result<MixerDeviceSink, DeviceSinkError>
462    where
463        E: FnMut(cpal::StreamError) + Send + 'static,
464    {
465        Self::validate_config(config);
466        let (controller, source) = mixer(config.channel_count, config.sample_rate);
467        Self::init_stream(device, config, source, error_callback).and_then(|stream| {
468            stream.play().map_err(DeviceSinkError::PlayError)?;
469            Ok(Self {
470                _stream: stream,
471                mixer: controller,
472                config: *config,
473                log_on_drop: true,
474            })
475        })
476    }
477
478    fn init_stream<S, E>(
479        device: &cpal::Device,
480        config: &DeviceSinkConfig,
481        mut samples: S,
482        error_callback: E,
483    ) -> Result<cpal::Stream, DeviceSinkError>
484    where
485        S: Source + Send + 'static,
486        E: FnMut(cpal::StreamError) + Send + 'static,
487    {
488        let cpal_config = config.into();
489
490        macro_rules! build_output_streams {
491            ($($sample_format:tt, $generic:ty);+) => {
492                match config.sample_format {
493                    $(
494                        cpal::SampleFormat::$sample_format => device.build_output_stream::<$generic, _, _>(
495                            &cpal_config,
496                            move |data, _| {
497                                data.iter_mut().for_each(|d| {
498                                    *d = samples
499                                        .next()
500                                        .map(Sample::from_sample)
501                                        .unwrap_or(<$generic>::EQUILIBRIUM)
502                                })
503                            },
504                            error_callback,
505                            None,
506                        ),
507                    )+
508                    _ => return Err(DeviceSinkError::UnsupportedSampleFormat),
509                }
510            };
511        }
512
513        let result = build_output_streams!(
514            F32, f32;
515            F64, f64;
516            I8, i8;
517            I16, i16;
518            I24, I24;
519            I32, i32;
520            I64, i64;
521            U8, u8;
522            U16, u16;
523            U24, cpal::U24;
524            U32, u32;
525            U64, u64
526        );
527
528        result.map_err(DeviceSinkError::BuildError)
529    }
530}

532/// Return all formats supported by the device.
533pub fn supported_output_configs(
534    device: &cpal::Device,
535) -> Result<impl Iterator<Item = cpal::SupportedStreamConfig>, DeviceSinkError> {
536    let mut supported: Vec<_> = device
537        .supported_output_configs()
538        .map_err(DeviceSinkError::SupportedConfigsError)?
539        .collect();
540    supported.sort_by(|a, b| b.cmp_default_heuristics(a));
541
542    Ok(supported.into_iter().flat_map(|sf| {
543        let max_rate = sf.max_sample_rate();
544        let min_rate = sf.min_sample_rate();
545        let mut formats = vec![sf.with_max_sample_rate()];
546        let preferred_rate = HZ_44100.get();
547        if preferred_rate < max_rate && preferred_rate > min_rate {
548            formats.push(sf.with_sample_rate(preferred_rate))
549        }
550        formats.push(sf.with_sample_rate(min_rate));
551        formats
552    }))
