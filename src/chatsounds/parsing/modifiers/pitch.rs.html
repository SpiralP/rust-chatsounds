1use nom::{branch::alt, bytes::complete::tag, IResult, Parser};
2use rodio::{buffer::SamplesBuffer, Source};
3
4use super::{parse_args, ModifierTrait};
5use crate::BoxSource;
6
7#[derive(Debug, PartialEq)]
8pub struct PitchModifier {
9    pub pitch: f32,
10}
11
12impl Default for PitchModifier {
13    fn default() -> Self {
14        Self { pitch: 1.0 }
15    }
16}
17
18impl ModifierTrait for PitchModifier {
19    fn parse(input: &str) -> IResult<&str, Self> {
20        // input = "pitch(2)"
21
22        let (input, _) = alt((tag("pitch"), tag("speed"))).parse(input)?;
23        let (input, args) = parse_args(input)?;
24
25        let mut modifier = Self::default();
26
27        if let Some(pitch) = args.first().copied().unwrap_or(None) {
28            // panics if 0
29            modifier.pitch = if (0.0..0.1).contains(&pitch) {
30                0.1
31            } else if pitch < 0.0 && pitch > -0.1 {
32                -0.1
33            } else {
34                pitch
35            };
36        }
37
38        Ok((input, modifier))
39    }
40
41    fn modify(&self, source: BoxSource) -> BoxSource {
42        if self.pitch >= 0.0 {
43            Box::new(source.speed(self.pitch))
44        } else {
45            let channels = source.channels();
46            let sample_rate = source.sample_rate();
47            let mut samples: Vec<_> = source.collect();
48            samples.reverse();
49
50            Box::new(SamplesBuffer::new(channels, sample_rate, samples).speed(self.pitch.abs()))
51        }
52    }
53}
